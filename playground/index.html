the jQuery

{% highlight javascript %} $.fn.sparallax = function(options) { var $window = $(window), defaults = { $elements: $(this), speed: 4 }, settings = $.extend(defaults, options); $.each(settings.$elements, function(i, el) { var $el = $(el), offset = 0, speed = ($el.data('parallax-speed') ? $el.data('parallax-speed') : defaults.speed); $window.scroll(function() { offset = -($window.scrollTop() / speed); $el.css({ backgroundPosition: '50% ' + offset + 'px' }); }); }); }; {% endhighlight %} {% highlight javascript %} $(document).ready(function() { $('.parallax').sparallax(); }); {% endhighlight %}

the css

{% highlight sass %} .parallax { background-repeat: no-repeat; background-position: 50% 0; -webkit-transition: background-position .2s ease-in; -moz-transition: background-position .2s ease-in; -o-transition: background-position .2s ease-in; transition: background-position .2s ease-in; -webkit-background-size: cover; background-size: cover; } {% endhighlight %}
